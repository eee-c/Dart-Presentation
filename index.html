Functional Dart

Javascript

function fib(i) {
  if (i < 2) return i;
  return fib(i-2) + fib(i-1);
}

Dart

fib(i) {
  if (i < 2) return i;
  return fib(i-2) + fib(i-1);
}

Calling Javascript Functions

function fib(i) {
  if (i < 2) return i;
  return fib(i-2) + fib(i-1);
}

fib(10); // => 55

Call Dart Functions

fib(i) {
  if (i < 2) return i;
  return fib(i-2) + fib(i-1);
}

fib(10); // => 55

Anonymous Functions in Javascript

function(i) {
  if (i < 2) return i;
  return fib(i-2) + fib(i-1);
}

Anonymous Functions in Dart

(i) {
  if (i < 2) return i;
  return fib(i-2) + fib(i-1);
}

Using Dart Functions

var list = [1, 5, 8, 10];
list.forEach((i) {fib_printer(i)});

fib_printer(i) {
  print("Fib($i): " + fib(i));
}

fib(i) {
  if (i < 2) return i;
  return fib(i-2) + fib(i-1);
}

Iterators

var list = [1, 5, 8, 10];
list.forEach((i) {fib_printer(i)});

Interpolating Strings

fib_printer(i) {
  print("Fib(${i}): " + fib(i));
}

fib_printer(10);
> Fib(10): 55

Shortcut #1

var list = [1, 5, 8, 10];
list.forEach((i) => fib_printer(i));

fib_printer(i) {
  print("Fib($i): " + fib(i));
}

fib(i) {
  if (i < 2) return i;
  return fib(i-2) + fib(i-1);
}

Shortcut #2

var list = [1, 5, 8, 10];
list.forEach(fib_printer);

fib_printer(i) {
  print("Fib($i): " + fib(i));
}

fib(i) {

  return fib(i-2) + fib(i-1);
}

First Order Support

add(x, y, z) {
  return x + y + z;
}

makeAdder1(fn, arg1) {
  return (y, z) {
    return fn(arg1, y, z);
  };
}

var add10 = makeAdder1(add, 10);
